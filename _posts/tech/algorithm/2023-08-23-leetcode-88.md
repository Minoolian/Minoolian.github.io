```java
public void merge(int[] nums1, int m, int[] nums2, int n) {
        int[] arr = new int[m + n]; // 1.

        int i = 0, j = 0, cur = 0;
        while (i<m && j<n) { // 2.
            if (nums1[i] <= nums2[j]){
                arr[cur] = nums1[i];
                i++;
            }else{
                arr[cur] = nums2[j];
                j++;
            }
            cur++;
        }

        if (i < m) { // 3.
            for (int k = cur; k < m+n; k++) {
                arr[k] = nums1[i];
                i++;
            }
        } else {
            for (int k = cur; k < m+n; k++) {
                arr[k] = nums2[j];
                j++;
            }
        }

        for (int k = 0; k < m+n; k++) {
            nums1[k] = arr[k];
        }
    }
```

```java
public void merge(int[] nums1, int m, int[] nums2, int n) {
        int i = m - 1;
        int j = n - 1;
        int k = m + n - 1;
        
        while (j >= 0) {
            if (i >= 0 && nums1[i] > nums2[j]) {
                nums1[k--] = nums1[i--];
            } else {
                nums1[k--] = nums2[j--];
            }
        }
    }
```
